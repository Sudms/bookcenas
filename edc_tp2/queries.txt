1 Conjunto de Predicados no Grafo
PREFIX mpred:<http://movies.org/pred/>
SELECT distinct ?p
WHERE{
	?s ?p ?o
}

2 Pesquisa de filmes realizados por um Realizador, dado o seu nome
PREFIX mpred:<http://movies.org/pred/>
SELECT ?movies
WHERE{
	?film mpred:directed_by ?who.
    	?who mpred:name ?name
    	FILTER regex(?name, "Peter Jackson", "i")
    	?film mpred:name ?movies
}

3 Pesquisa de atores com participação num filme, dado o seu nome
PREFIX mpred:<http://movies.org/pred/>
SELECT ?movies
WHERE{
	?film mpred:starring ?who.
    	?who mpred:name ?name
    	FILTER regex(?name, "Kim Novak", "i")
    	?film mpred:name ?movies
}

4 Descrição total de um filme, dado o seu nome
PREFIX mpred:<http://movies.org/pred/>
SELECT ?p ?o
WHERE{
    	?film mpred:directed_by ?director.
    	?film mpred:name ?s
    	FILTER regex(?s, "I shot Andy Warhol", "i").
    	?film ?p ?o.
}

5 Inferência do tipo filme (rdf:type Movie)
PREFIX mpred:<http://movies.org/pred/>
SELECT ?s
WHERE{
    	?film mpred:directed_by ?director.
    	?film mpred:name ?s
}
